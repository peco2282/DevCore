## Install (Gradle Kotlin DSL)

  implementation("com.peco2282.devcore:command")
  implementation("com.peco2282.devcore:config")
  implementation("com.peco2282.devcore:scheduler")
  implementation("com.peco2282.devcore:cooldown")
}
